⚡ Vue3 组件化实现 Composition API 响应式数据 路由集成

<!-- Navbar.vue -->
      <template>
        <nav class="navbar" role="navigation" aria-label="主导航">
          <router-link to="/" class="brand">🚀 Logo</router-link>

          <ul :class="['nav-links', { active: isMenuOpen }]">
            <li v-for="item in menuItems" :key="item.path">
              <router-link :to="item.path"
                @click="closeMenu">
                {{ item.text }}
              </router-link>
            </li>
          </ul>

          <button
            class="hamburger"
            @click="toggleMenu"
            :aria-expanded="isMenuOpen"
            aria-controls="nav-menu"
            aria-label="切换菜单"
          >
            <span /><span /><span />
          </button>
        </nav>
      </template>
// Navbar.vue — Composition API
      import { ref, watch, onMounted, onUnmounted } from 'vue';
      import { useRoute } from 'vue-router';

      export default {
        setup() {
          const isMenuOpen = ref(false);
          const route = useRoute();

          const menuItems = [
            { path: '/',       text: '首页' },
            { path: '/products', text: '产品' },
            { path: '/docs',     text: '文档' },
            { path: '/about',    text: '关于' },
          ];

          const toggleMenu = () => { isMenuOpen.value = !isMenuOpen.value; };
          const closeMenu = () => { isMenuOpen.value = false; };

          // 窗口resize时自动关闭菜单
          const onResize = () => {
            if (window.innerWidth > 768 && isMenuOpen.value) closeMenu();
          };
          onMounted(() => window.addEventListener('resize', onResize));
          onUnmounted(() => window.removeEventListener('resize', onResize));

          // 路由切换时关闭菜单
          watch(() => route.path, closeMenu);

          return { isMenuOpen, menuItems, toggleMenu, closeMenu };
        }
      };
/* Navbar 样式 — Mobile-First */
      .navbar {
        display: flex; justify-content: space-between; align-items: center;
        padding: 0 1.2rem; height: 56px; background: #1e293b; color: white;
      }
      .nav-links { display: flex; gap: 1.5rem; list-style: none; }

      /* 移动端 <768px */
      @media (max-width: 768px) {
        .nav-links {
          position: absolute; left: 0; right: 0; top: 56px;
          background: #1e293b; flex-direction: column;
          max-height: 0; overflow: hidden; transition: max-height .35s;
        }
        .nav-links.active { max-height: 400px; }
        .nav-links a {
          display: block; padding: 1rem 1.5rem; /* 16px padding */
          min-height: 44px;
        }
      }
