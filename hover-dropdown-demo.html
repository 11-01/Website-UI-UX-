📝 进阶方案核心CSS（推荐套用）

<!-- 语义化结构：nav > ul > li > a + ul.submenu -->
<nav>
  <ul class="main-menu">
    <li class="menu-item">
      <a href="#">首页</a>
    </li>
    <li class="menu-item">
      <a href="#">产品中心 ▾</a>
      <!-- 子菜单嵌套在父li中 -->
      <ul class="submenu">
        <li><a href="#">软件产品</a></li>
        <li><a href="#">硬件设备</a></li>
        <li><a href="#">解决方案</a></li>
      </ul>
    </li>
    <li class="menu-item">
      <a href="#">联系我们</a>
    </li>
  </ul>
</nav>
/* ===== 核心三板斧 ===== */
.menu-item { position: relative; }  /* 为子菜单的 absolute 提供定位基准 */

.submenu {
  position: absolute;
  top: 100%;           /* 紧贴父菜单下方 */
  left: 0;
  min-width: 170px;
  background: white;
  box-shadow: 0 12px 40px rgba(0,0,0,.12);
  z-index: 1000;

  /* ✅ 过渡三件套（替代display:none） */
  opacity: 0;
  visibility: hidden;
  transform: translateY(-8px);
  transition: opacity .25s ease,
              visibility .25s ease,
              transform .25s cubic-bezier(.4,0,.2,1);
}

/* hover 触发 */
.menu-item:hover .submenu {
  opacity: 1;
  visibility: visible;
  transform: translateY(0);    /* 下滑到原位 */
}

/* 箭头动画 */
.menu-item > a .arrow {
  transition: transform .3s ease;
}
.menu-item:hover > a .arrow {
  transform: rotate(180deg);   /* 箭头旋转180° */
}
/* ❌ 方案A：display:none → 无法过渡 */
.submenu {
  display: none;             /* 瞬间隐藏 */
}
.menu-item:hover .submenu {
  display: block;            /* 瞬间出现 — 无法添加transition */
}
/* 可加过渡吗？ transition 对 display 无效 🙅 */

/* ✅ 方案B：opacity+visibility+transform — 完美过渡 */
.submenu {
  opacity: 0;
  visibility: hidden;       /* 隐藏时不可交互 */
  transform: translateY(-8px); /* 初始向上偏移 */
  transition: opacity .25s, visibility .25s, transform .25s;
}
.menu-item:hover .submenu {
  opacity: 1;
  visibility: visible;      /* 可交互 */
  transform: translateY(0);
}
/* 为什么用visibility而不用opacity:0？            */
/* opacity:0 的元素仍然可以被鼠标触发hover        */
/* visibility:hidden 彻底阻止交互，体验更好       */
