/* === ж–№жЎҲдёҖпјҡзәҜ CSSпјҢ3иЎҢжҗһе®ҡ === */
.navbar {
  position: sticky;
  top: 0;
  z-index: 1000;
  background-color: white;     /* еҝ…йЎ»пјҒеҗҰеҲҷеҶ…е®№йҖҸеҮә */
  box-shadow: 0 2px 5px rgba(0,0,0,.1);
}

/* вҡ пёҸ зЎ®дҝқжүҖжңүзҘ–е…Ҳе…ғзҙ жІЎжңү overflow: hidden */
/* еҗҰеҲҷ position:sticky дјҡеӨұж•ҲпјҒ */
/* === ж–№жЎҲдәҢпјҡJSзӣ‘еҗ¬ + CSS class === */
// JS вҖ” йңҖиҠӮжөҒпјҲзӨәдҫӢз®ҖеҢ–пјү
window.addEventListener('scroll', () => {
  const navbar = document.querySelector('.navbar');
  if (window.scrollY > 100) {
    navbar.classList.add('sticky');
  } else {
    navbar.classList.remove('sticky');
  }
});

/* CSS */
.navbar { transition: all 0.35s ease; }
.navbar.sticky {
  position: fixed; top: 0; width: 100%;
  background: white;
  box-shadow: 0 2px 16px rgba(0,0,0,.1);
}
/* еҶ…е®№еҢәйңҖиЎҘ padding-top йҒҝе…ҚдёҠз§» */
body { padding-top: 56px; }
/* === ж–№жЎҲдёүпјҡIntersection Observer === */
// HTML дёӯж”ҫдёҖдёӘе“Ёе…өе…ғзҙ 
// <div class="sentinel"></div> вҖ” еңЁеҜјиҲӘж ҸдёҠж–№

// JS вҖ” ж— йңҖиҠӮжөҒпјҢжөҸи§ҲеҷЁеҺҹз”ҹй«ҳж•Ҳ
const observer = new IntersectionObserver(
  ([entry]) => {
    const navbar = document.querySelector('.navbar');
    if (!entry.isIntersecting) {
      navbar.classList.add('sticky');
    } else {
      navbar.classList.remove('sticky');
    }
  },
  { threshold: 0.1 }
);
observer.observe(document.querySelector('.sentinel'));

.banner { height: 400px; }    /* Bannerжңүи¶іеӨҹй«ҳеәҰ */
.navbar {
  position: sticky;           /* зӣёеҜ№зҲ¶е®№еҷЁBanner */
  top: 0;                    /* еҲ°иҫҫзҲ¶е®№еҷЁеә•йғЁж—¶еҗёйҷ„ */
  background: white;         /* еҝ…йЎ»и®ҫзҪ®иғҢжҷҜиүІ */
  z-index: 100;
}

/* JSпјҡеҗёйЎ¶ж—¶еҠЁжҖҒж”№еҸҳж ·ејҸпјҲз”ЁgetBoundingClientRectжӣҙзІҫеҮҶпјү */
window.addEventListener('scroll', () => {
  const banner = document.querySelector('.banner');
  const navbar = document.querySelector('.navbar');
  const bannerBottom = banner.getBoundingClientRect().bottom;
  navbar.classList.toggle('active', bannerBottom <= 0);
});
