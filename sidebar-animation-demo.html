📝 三种方案核心代码对比
/* ===== 方案一：width 过渡 ===== */
.sidebar { width: 220px; transition: width .35s ease; overflow: hidden; }
.sidebar.collapsed { width: 64px; }
/* ⚠️ 触发Layout重排，内容会被挤压换行 */

/* ===== 方案二：translateX（推荐✅）===== */
.sidebar {
  width: 220px; transition: transform .35s ease;
  will-change: transform;   /* 提前通知浏览器优化 */
}
.sidebar.collapsed { transform: translateX(-220px); }
/* ✅ 仅触发Composite，GPU加速，60fps */
/* 收起时文字淡化 */
.sidebar.collapsed .sidebar-content { opacity: 0; }

/* ===== 方案三：Grid联动 ===== */
.container {
  display: grid;
  grid-template-columns: 220px 1fr;
  transition: grid-template-columns .35s ease;
}
.container.collapsed { grid-template-columns: 64px 1fr; }
/* 主内容区自动联动，无需手动margin-left */

/* ===== 通用 JS ===== */
toggleBtn.addEventListener('click', () => {
  sidebar.classList.toggle('collapsed');
});
